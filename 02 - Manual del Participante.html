    buenas <- input$inspeccionadas - input$defectuosas

  })

    paste0(round(fpy_valor(), 2), "%")

      tipo = c("Buenas", "Defectuosas"),

                   input$defectuosas)

      geom_bar(stat = "identity") +
      scale_fill_manual(values = c("Buenas" = "#28a745", 
                                   "Defectuosas" = "#dc3545")) +
      theme_minimal() +
      labs(title = paste("Turno:", input$turno))
  })
